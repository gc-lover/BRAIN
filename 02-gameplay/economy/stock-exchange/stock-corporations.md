### Corporation Entity

```sql
CREATE TABLE stock_corporations (
    id VARCHAR(100) PRIMARY KEY,
    ticker VARCHAR(10) UNIQUE NOT NULL, -- "ARSK", "MLTC"
    name VARCHAR(200) NOT NULL,
    sector VARCHAR(100) NOT NULL,
    country VARCHAR(100) NOT NULL,
    
    -- Financial data
    market_cap BIGINT NOT NULL,
    shares_outstanding BIGINT NOT NULL,
    public_float_percent DECIMAL(5,2) NOT NULL,
    
    -- Current state
    is_listed BOOLEAN DEFAULT TRUE,
    listing_tier VARCHAR(20) NOT NULL, -- "TIER1", "TIER2", "TIER3", "GROWTH"
    listed_since DATE NOT NULL,
    delisted_at DATE,
    delisting_reason TEXT,
    
    -- Company info
    ceo_npc_id VARCHAR(100), -- May change through quests
    headquarters VARCHAR(100),
    description TEXT,
    
    -- M–µ—Ç–∞
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    
    CONSTRAINT fk_corporation_ceo FOREIGN KEY (ceo_npc_id) REFERENCES npcs(id)
);

CREATE INDEX idx_stock_corporations_listed ON stock_corporations(is_listed) WHERE is_listed = TRUE;
CREATE INDEX idx_stock_corporations_tier ON stock_corporations(listing_tier);
CREATE INDEX idx_stock_corporations_sector ON stock_corporations(sector);
```

### Stock Prices (Current)

```sql
CREATE TABLE stock_prices (
    id SERIAL PRIMARY KEY,
    corporation_id VARCHAR(100) NOT NULL,
    
    -- Price
    current_price DECIMAL(12,2) NOT NULL,
    previous_close DECIMAL(12,2) NOT NULL,
    
    -- Daily stats
    day_open DECIMAL(12,2) NOT NULL,
    day_high DECIMAL(12,2) NOT NULL,
    day_low DECIMAL(12,2) NOT NULL,
    day_volume BIGINT NOT NULL,
    
    -- Changes
    change_amount DECIMAL(12,2) NOT NULL, -- current - previous_close
    change_percent DECIMAL(5,2) NOT NULL,
    
    -- 52-week
    week_52_high DECIMAL(12,2) NOT NULL,
    week_52_low DECIMAL(12,2) NOT NULL,
    
    -- Timestamp
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    
    CONSTRAINT fk_price_corporation FOREIGN KEY (corporation_id) REFERENCES stock_corporations(id) ON DELETE CASCADE,
    UNIQUE(corporation_id)
);

CREATE INDEX idx_stock_prices_updated ON stock_prices(updated_at);
```
